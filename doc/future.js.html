Params
{Number} duration Number of milliseconds to reject at
Returns
Future The instremented future 
Future.prototype.ready = function(duration) {
  var future = new Future();
  
  this.then(function(value) {
    future.fulfill(value);
  });

  setTimeout(function() {
    future.reject(new Error('Did not resolve future in ' + duration + ' ms'));
  }, duration);

  return future;
};

Future.prototype.fallbackTo = function(promise) {
  var future = new Future();

  this.then(function(value) {
    future.fulfill(value);
  }, function() {
    promise.then(function(value) {
      future.fulfill(value);
    });
  });

  return future;
};

Params
{Function} predicate The predicate to test the value of the future
Returns
Future The new future 
Future.prototype.filter = function(predicate) {
  var future = new Future();

  this.then(function(value) {
    if (predicate(value)) {
      return future.fulfill(value);
    }

    return future.reject(new Error('Value does not match predicate'));
  }, function(reason) {
    future.reject(reason);
  });

  return future;
};

Takes two futures and zips their values as a tuple

Params
{Promise} promise The promise to be applied on the right of the tuple
Returns
Future The new future which will fulfill with the tuple 
Future.prototype.zip = function(promise) {
  var future = new Future();

  this.then(function(left) {
    promise.then(function(right) {
      future.fulfill([left, right]);
    });
  });

  return future;
};
